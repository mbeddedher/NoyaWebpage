CREATE TABLE prices (
    id SERIAL PRIMARY KEY, -- Unique identifier for price entry
    product_id INT NOT NULL, -- Foreign key to reference the product
    price DECIMAL(10, 2) NOT NULL, -- The price value

    is_multi BOOLEAN DEFAULT false,
	is_linked BOOLEAN DEFAULT FALSE,
   
	is_component BOOLEAN DEFAULT FALSE,
	cost DECIMAL(10, 2) NOT NULL,

-- Create a function to validate multi_currency_prices insert
CREATE OR REPLACE FUNCTION validate_multi_currency_price()
RETURNS TRIGGER AS $$
BEGIN
    IF NOT EXISTS (
        SELECT 1 FROM prices 
        WHERE id = NEW.price_id 
        AND is_multi = true
    ) THEN
        RAISE EXCEPTION 'Cannot add multi-currency price for non-multi-currency price record';
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Create trigger to validate multi_currency_prices insert
CREATE TRIGGER check_multi_currency_price
BEFORE INSERT ON multi_currency_prices
FOR EACH ROW
EXECUTE FUNCTION validate_multi_currency_price();
